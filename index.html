-- Roblox Studio Plugin
-- Enable: Studio Settings > Security > Allow HTTP Requests

local HttpService = game:GetService("HttpService")
local Selection = game:GetService("Selection")

local ENDPOINT = "http://localhost:5500/ai"

local toolbar = plugin:CreateToolbar("AI Assistant")
local fixBtn  = toolbar:CreateButton("Fix Script", "Fix selected script with AI", "rbxassetid://6035067966")
local genBtn  = toolbar:CreateButton("Generate",   "Generate a script from prompt", "rbxassetid://6035067966")

local function getSelectedScript()
    local sel = Selection:Get()
    if #sel == 0 then return nil end
    local obj = sel[1]
    if obj:IsA("LuaSourceContainer") then
        return obj
    end
    return nil
end

local function callAI(prompt, code)
    local ok, resp = pcall(function()
        return HttpService:PostAsync(
            ENDPOINT,
            HttpService:JSONEncode({ prompt = prompt, code = code or "" }),
            Enum.HttpContentType.ApplicationJson
        )
    end)
    if not ok then
        warn("[AI Assistant] Request failed: " .. tostring(resp))
        return nil
    end
    local data = HttpService:JSONDecode(resp)
    return data.result
end

fixBtn.Click:Connect(function()
    local scr = getSelectedScript()
    if not scr then
        warn("[AI Assistant] Select a Script first.")
        return
    end
    local result = callAI("Fix this Luau script. Return only corrected code.", scr.Source)
    if result then
        scr.Source = result
        print("[AI Assistant] Script fixed!")
    end
end)

genBtn.Click:Connect(function()
    local prompt = "Create a basic Roblox game system"
    -- Replace with your own prompt input UI if desired
    local result = callAI(prompt, "")
    if result then
        local newScript = Instance.new("Script")
        newScript.Source = result
        newScript.Parent = game.ServerScriptService
        newScript.Name  = "AI_Generated"
        Selection:Set({newScript})
        print("[AI Assistant] Script generated!")
    end
end)
