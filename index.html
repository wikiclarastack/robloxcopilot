Setup Guide

          

            

              
1

              

                Enable HTTP Requests
                
Studio → File → Studio Settings → Security → Allow HTTP Requests

              

            

            

              
2

              

                Add your API Key
                
Go to Settings tab, paste your Groq key and click Save.

              

            

            

              
3

              

                Create Plugin Script
                
In Studio, create a Script in a Plugin folder. Paste the code shown.

              

            

            

              
4

              

                Update the URL
                
Replace your-site.netlify.app with your actual Netlify subdomain.

              

            

          

          

            
            Since this site runs on Netlify (no local server), the plugin sends requests to your deployed URL. No live-server needed.

-- ╔══════════════════════════╗
-- ║  Luau Studio Plugin 1.0  ║
-- ╚══════════════════════════╝
-- Enable: Studio Settings > Security > Allow HTTP Requests

local HttpService = game:GetService("HttpService")
local Selection   = game:GetService("Selection")

-- 🔧 Replace with your Netlify URL
local ENDPOINT = "https://your-site.netlify.app/api/ai"

local toolbar    = plugin:CreateToolbar("Luau Studio AI")
local btnFix     = toolbar:CreateButton("✦ Fix",      "Fix selected script",   "rbxassetid://6035067966")
local btnOptimize= toolbar:CreateButton("⚡ Optimize","Optimize script",        "rbxassetid://6035067966")
local btnExplain = toolbar:CreateButton("? Explain",  "Explain in output",     "rbxassetid://6035067966")

local function getScript()
    local sel = Selection:Get()
    if #sel == 0 then warn("[AI] Select a script first.") return nil end
    local obj = sel[1]
    if obj:IsA("LuaSourceContainer") then return obj end
    warn("[AI] Not a script object.") return nil
end

local function callAI(action, code, prompt)
    local ok, res = pcall(function()
        return HttpService:PostAsync(
            ENDPOINT,
            HttpService:JSONEncode({
                action = action,
                code   = code,
                prompt = prompt or ""
            }),
            Enum.HttpContentType.ApplicationJson
        )
    end)
    if not ok then warn("[AI] Failed: " .. tostring(res)) return nil end
    local data = HttpService:JSONDecode(res)
    return data.result
end

btnFix.Click:Connect(function()
    local s = getScript() if not s then return end
    print("[AI] Fixing…")
    local r = callAI("fix", s.Source, "Fix all bugs")
    if r then s.Source = r print("[AI] ✅ Fixed!") end
end)

btnOptimize.Click:Connect(function()
    local s = getScript() if not s then return end
    print("[AI] Optimizing…")
    local r = callAI("optimize", s.Source, "Optimize performance")
    if r then s.Source = r print("[AI] ✅ Optimized!") end
end)

btnExplain.Click:Connect(function()
    local s = getScript() if not s then return end
    print("[AI] Analyzing…")
    local r = callAI("explain", s.Source, "")
    if r then
        print("── AI Explanation ──────────────")
        print(r)
        print("────────────────────────────────")
    end
end)
